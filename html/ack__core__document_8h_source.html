22// Element kinds.
23//
24// MUST fit into 4 bits, because as an important optimization we pack keys with values [1-15] into
25// the high nibble of a type byte for object members. If we need more types, we can use
26// ACK_DOC_ELEMENT_ESCAPE to indicate the type, and modify the member header parser and emitter in
27// ack_core_document.c to understand that when it sees ACKDOC_ELEMENT_ESCAPE as the type, it should load
28// the actual type from a subsequent byte.
29//
30// Also note that ACKDOC_ELEMENT_NONE is not actually used in documents, but is defined here for
31// convenience.
32#define ACKDOC_ELEMENT_NONE 0 // Not actually used in documents
33#define ACKDOC_ELEMENT_OBJECT 1
34#define ACKDOC_ELEMENT_ARRAY 2
35#define ACKDOC_ELEMENT_STRING 3 // value: bytes of string, including terminating nul
36#define ACKDOC_ELEMENT_INTEGER 4 // value: variable-length encoded
37#define ACKDOC_ELEMENT_FLOAT64 5 // value: 8 bytes representing an IEEE754 64-bit float
38#define ACKDOC_ELEMENT_TRUE 6
39#define ACKDOC_ELEMENT_FALSE 7
40#define ACKDOC_ELEMENT_NULL 8
41#define ACKDOC_ELEMENT_CONTAINER_END 9
42#define ACKDOC_ELEMENT_NEGATIVE_INTEGER 10 // value: same as integer, bits flipped (for negative numbers)
43#define ACKDOC_ELEMENT_FLOAT32 11 // value: 4 bytes representing an IEEE754 32-bit float
44#define ACKDOC_ELEMENT_RESERVED_3 12
45#define ACKDOC_ELEMENT_RESERVED_4 13
46#define ACKDOC_ELEMENT_RESERVED_5 14
47#define ACKDOC_ELEMENT_ESCAPE 15 // Reserved; not currently used
