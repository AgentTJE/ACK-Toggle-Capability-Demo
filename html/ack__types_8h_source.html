34// Copy strings in a way that ensures nul termination.
35// The target is intended to be a fixed array of characters, and truncation occurs if necessary.
36#define ACK_SAFE_COPY_STRING(target, source) \
37 if (sizeof(target)) \
38 { \
39 if (source) \
40 { \
41 strncpy(target, source, sizeof(target) - 1); \
42 target[sizeof(target) - 1] = '\0'; \
43 } \
44 else \
45 { \
46 target[0] = '\0'; \
47 } \
48 }

89// In a ChangeReport event, distinguishes a context property from a payload property.
90// If set, the property is payload, e.g., associated with a change due to some local cause such as the user
91// turning on a light switch. If clear, the property is context in a ChangeReport, e.g. the value of some
92// other property on the device but not the one that changed.
93// See https://developer.amazon.com/docs/device-apis/alexa-interface.html#changereport for more info.
94// Ignored for non-ChangeReport events (plain responses, error responses, state reports).
